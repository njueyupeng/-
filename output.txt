$(function() {
    var h = new Common.home();
    var nav = new Common.navbar();

    var placeholder = '请输入您要搜索的内容';

    //点击埋点.
    $("a[name^=ysc_sy_]").live("click", function () {
        try {
            sa.click.sendDatasIndex(this);
        } catch (e) {
        }
    });


    $.post("http://www.suningcloud.com/ajax/getHotKeywords.htm", function (data) {
        if (data) {
            placeholder = data[0];
            $('#keyWord').attr('placeholder',placeholder);
        }
    });

    //我的代码
    var h = true;
    $('.j_searchbox').focus(function () {
        h = true;
        $(this).attr('placeholder', '');
        $.post("http://www.suningcloud.com/ajax/getHotKeywords.htm", function (data) {
            if (data) {
                $('.search-word li').unbind('click');

                var rows = ['<li class="noBg">大家正在搜 &gt;</li>'];
                $.each(data, function (n, val) {
                    var li = '<li>' + val + '</li>';
                    rows.push(li);
                });
                $('#hotWordUl').html(rows.join(''));

                $('.search-word li').click(function(){
                    if($(this).index()!=0){
                        var str=$(this).text();
                        $('.j_searchbox').val(str);
                        $('.search-word').hide();
                        var url = "http://www.suningcloud.com/searchAll/searchByKeyWord.htm?keyWord=" + encodeURIComponent(str);
                        window.location.href = encodeURI(url);
                    }
                });
            }
        });
        $('.search-word').show();
    });

    $('.j_searchbox,.search-word').focusout(function(ev){
        if($(this).val()==''){
            $(this).attr('placeholder',placeholder);
        }
        if(h){
            $('.search-word').hide();
        }
    }).mouseover(function(){
        h=false;

    }).mouseout(function(){
        h=true;
    });

});

</script>
